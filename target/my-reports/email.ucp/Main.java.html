package email.ucp;

import java.util.Arrays;

public class Main {
    public static void main(String[] args) {
        // Crear buzones para el remitente y destinatario
        Buzon buzonRemitente = new Buzon();
        Buzon buzonDestinatario = new Buzon();

        // Crear los contactos
        Contacto remitente = new Contacto("Sebastiano Corradi", "sebastianocorradi57@gmail.com");
        Contacto destinatario = new Contacto("Maria Sanchez", "maria@example.com");

        // Crear un correo
        Correo correo = new Correo("Mata", "Hello", remitente, Arrays.asList(destinatario));

        // Agregar el correo a la bandeja de enviados del remitente
        buzonRemitente.agregarAEnviados(correo);

        // Enviar el correo desde el remitente al destinatario
        EnviarCorreo.enviar(correo, buzonRemitente, buzonDestinatario);

        // Verificar que el correo está en la bandeja de entrada del destinatario
        System.out.println("Bandeja de entrada del destinatario:");
        for (Correo c : buzonDestinatario.getBandejaEntrada()) {
            System.out.println("Asunto: " + c.getAsunto() + ", Remitente: " + c.getRemitente().getNombre());
        }

        // Verificar que el correo ya no está en la bandeja de enviados del remitente
        System.out.println("Bandeja de enviados del remitente:");
        for (Correo c : buzonRemitente.getBandejaEnviados()) {
            System.out.println("Asunto: " + c.getAsunto() + ", Remitente: " + c.getRemitente().getNombre());
        }
    }
}
